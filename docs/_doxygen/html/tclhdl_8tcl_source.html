242 #-------------------------------------------------------------------------------
243 ## Add IP To The Project Structure
244 #
245 #-------------------------------------------------------------------------------
246 proc ::tclhdl::add_ip {type src} {
247  switch $::tclhdl::project_tool {
248  INTEL_QUARTUS {
249  switch $type {
250  INTEL_QUARTUS_IP {
251  if { $::tclhdl::flag_project_create == 1} {
252  log::log debug "add_ip: Adding ip type $type - $src"
253  cd $::tclhdl::quartus::output_root
254  ::tclhdl::quartus::ip_reset
255  ::tclhdl::quartus::ip_set_type $type
256  source $src
257  ::tclhdl::quartus::ip_generate
258  }
259  }
260  INTEL_QUARTUS_QSYS {
261  if { $::tclhdl::flag_project_create == 1} {
262  log::log debug "add_ip: Adding QSYS ip type $type - $src"
263  ::tclhdl::quartus::ip_reset
264  ::tclhdl::quartus::ip_set_type $type
265  source $src
266 
267  log::log debug "add_ip: Adding QSYS ip $::tclhdl::quartus::output_root"
268  set ip_name "$::tclhdl::quartus::output_root/$::tclhdl::quartus::ip_name"
269  file mkdir $ip_name
270  cd $ip_name
271 
272  foreach dir $::tclhdl::quartus::qsys_component {
273  log::log debug "add_ip: Adding QSYS component - $dir"
274  set name [file tail $dir]
275  file link -symbolic $name $dir
276  }
277 
278  foreach hw $::tclhdl::quartus::qsys_hw {
279  log::log debug "add_ip: Adding QSYS hw - $hw"
280  set name [file tail $hw]
281  file copy -force $hw "$ip_name/$name"
282  }
283 
284  ::tclhdl::quartus::ip_generate
285  }
286  }
287  default {
288  log::logMsg "add_ip: No IP type defined for $ip_file"
289  }
290  }
291  }
292  XILINX_VIVADO {
293  switch $type {
294  XCI {
295  log::log debug "add_ip: Adding ip type $type - $src"
296  ::tclhdl::vivado::ip_add $type $src
297  }
298  default {
299  log::logMsg "add_ip: No IP type defined for $ip_file"
300  }
301  }
302  }
303  XILINX_ISE {
304  switch $type {
305  COREGEN {
306  log::log debug "add_ip: Adding ip type $type - $src"
307  ::tclhdl::ise::ip_add $type $src
308  }
309  XCO {
310  log::log debug "add_ip: Adding ip type $type - $src"
311  ::tclhdl::ise::ip_add $type $src
312  }
313  default {
314  log::logMsg "add_ip: No IP type defined for $ip_file"
315  }
316  }
317  }
318  LATTICE_DIAMOND {
319  switch $type {
320  LATTICE_DIAMOND_IP {
321  log::log debug "add_ip: Adding ip type $type - $src"
322  source $src
323  }
324  default {
325  log::logMsg "add_ip: No IP type defined for $ip_file"
326  }
327  }
328  }
329  default {
330  log::logMsg "add_source: No supported tool define for the current project"
331  }
332  }
333  #-- Make sure we are always at project build dir
334  cd $::tclhdl::project_build_dir
335 }

483 ## Verify if Project is compliant with TCLHDL
484 #
485 #------------------------------------------------------------------------------
486 proc ::tclhdl::project_verify {prj} {
487  global ::tclhdl::project_dir
488  global ::tclhdl::project_target_dir
489 
490  log::log debug "project_verify: Verify project $prj at root $tclhdl::project_root"
491  set project_dir $::tclhdl::project_root
492  set target_dir $project_dir
493 
494  log::log debug "project_verify: Check if exists $target_dir"
495  if { ![file exist $target_dir] } {
496  log::logMsg "project_verify: Project $target_dir does not exist"
497  exit 1
498  }
499 
500  log::log debug "project_verify: Check if structure exists at $target_dir/project"
501  if {![file exist $target_dir/project] && ![file exist $target_dir/sources] && ![file exist $target_dir/build] && ![file exist $target_dir/ip] } {
502  log::logMsg "project_verify: Project $prj is not compliant with TCLHDL"
503  exit 1
504  }
505 
506  set ::tclhdl::project_dir [file normalize $project_dir]
507  set ::tclhdl::project_target_dir [file normalize $target_dir]
508 }

510 #-------------------------------------------------------------------------------
511 ## Create/Generate Project
512 #
513 #-------------------------------------------------------------------------------
514 proc ::tclhdl::project_create {prj} {
515  global ::tclhdl::flag_project_create
516  global ::tclhdl::project_build_dir
517  global ::tclhdl::project_build_ip_dir
518 
519  #-- Verify Project
520  log::log debug "project_create: Verifying if project exists"
521  ::tclhdl::project_verify $prj
522 
523  #-- Get Environment Variables
524  log::log debug "project_create: Setting environment"
525  set ::tclhdl::flag_project_create 0
526  source $::tclhdl::project_target_dir/project
527  set ::tclhdl::flag_project_create 1
528 
529  #-- Create Project
530  log::log debug "project_create: Creating project"
531 
532  foreach lst $::tclhdl::list_projects {
533  set project [lindex $lst 0]
534  set settings [lindex $lst 1]
535  set current_dir [pwd]
536 
537  #-- Create necessary directories
538  file delete -force "$::tclhdl::project_target_dir/$project-$settings"
539  file mkdir "$::tclhdl::project_target_dir/$project-$settings"
540  file mkdir "$::tclhdl::project_target_dir/$project-$settings/ip"
541 
542  set ::tclhdl::project_build_dir [file normalize "$::tclhdl::project_target_dir/$project-$settings"]
543  set ::tclhdl::project_build_ip_dir [file normalize "$::tclhdl::project_build_dir/ip"]
544 
545  #-- Creating Projects on the proper right folder
546  log::log debug "project_create: Opening Project with $settings"
547  cd "$::tclhdl::project_target_dir/$project-$settings"
548  ::tclhdl::project_open [lindex $lst 0] [lindex $lst 1] [lindex $lst 2]
549 
550  log::log debug "project_create: Reload Project with $settings"
551  cd $current_dir
552  source $::tclhdl::project_target_dir/settings
553  source $::tclhdl::project_target_dir/project
554 
555  log::log debug "project_create: Close Project with $settings"
556  ::tclhdl::project_close [lindex $lst 0]
557  }
558 }
